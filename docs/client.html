  $commandline.addEventListener 'keyup', (e) ->
    switch e.keyIdentifier
      when 'Enter'
        socket.send({ 'command': $commandline.value })
        $history.innerHTML += '<span class="command">' + $commandline.value + '</span>\n'
        $commandline.value = ''
      when 'Up'
        socket.send({ 'up' })
      when 'Down'
        socket.send({ 'down' })
      when 'Left'
        socket.send({ 'left' })
      when 'Right'
        socket.send({ 'right' })

    switch e.keyCode
      when 67
        socket.send({ 'kill' }) if e.ctrlKey
